175def burnt_percentage(universe : numpy.array): 

176 '''cette fonction prend en argument un univers et affiche le taux de végétation brûlée en haut à gauche de l'écran''' 

177 nb_burnt = 0 

178 nb_brulure_possible = 0 

179 # calcul du pourcentage

181 for j in range(len(universe[0])): 

182 if not isinstance(universe[i,j], Water): 

183 nb_brulure_possible += universe[i,j].lifetime + 1 

184 nb_burnt += universe[i,j].time
